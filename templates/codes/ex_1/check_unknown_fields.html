Здесь в качестве аргумента из метода add_player_to_csv() приходит список данных, введенных пользователем. Циклом каждый элемент списка проверяется на пустоту, если условие выполнено, элемент заменяется на строку "Неизвестно" – иначе продолжаем перебор. В конце возвращается список отформатированных элементов. 

@staticmethod 
def check_unknown_fields(player_info: list): 
    for index, field in enumerate(player_info): 
        if field == '' or len(field) == 2: 
            player_info[index] = 'Неизвестно' 
        continue 
    return player_info
