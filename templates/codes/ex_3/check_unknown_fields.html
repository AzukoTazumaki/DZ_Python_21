Метод проверки на пустоту

Полностью аналогичен методу из задания с баскетболистами, за исключением названия переменных. Возвращает отформатированный список. 


@staticmethod 
def check_unknown_fields(employee_info: list) -> list: 
    for index, field in enumerate(employee_info): 
        if len(re.sub(r'\s+', '', field)) == 0: 
            employee_info[index] = 'Неизвестно' 
        continue 
    return employee_info
